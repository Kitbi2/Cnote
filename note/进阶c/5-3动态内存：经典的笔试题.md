# 5-3动态内存：经典的笔试题

下面代码运行的结果是什么

```c
#include<stdio.h>
void getmemory(char *p)
{
	p = (char *)malloc(100);
}
void test()
{
	char *ptr = NULL;
	getmemory(ptr);
	strcpy(str,"sha bi");
	printf(str);
}
int main()
{
	test();
	return 0;
}
```

```c
#include<stdio.h>
char * Getmemory(void)
{
	char p[] = "hello world";
	return p;
}
void Test(void)
{
	char *str = NULL;
	str = Getmemory();
	printf(str);
}
int main()
{
	test();

}
```
